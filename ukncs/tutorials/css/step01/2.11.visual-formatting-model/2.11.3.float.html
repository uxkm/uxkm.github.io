float 해제 3가지 방법

부모요소 하단에 빈 요소 삽입하여 clear:both

<style>
	/* float 해제 빈 요소 삽입 */
	.parent{
		border:4px solid #000;
	}
	.child:nth-child(1){
		float:left;
		width:45%;
		height:50px;
		background-color:gold;
	}
	.child:nth-child(2){
		float:right;
		width:45%;
		height:50px;
		background-color:yellowgreen;
	}
	.clear{
		clear:both;
	}
	.footer{
		height:40px;
		background-color:brown;
	}
</style>
<div class="parent">
	<div class="child">좌측 정렬</div>
	<div class="child">우측 정렬</div>
	<div class="clear"></div>
</div>
<div class="footer">하단 정렬</div>
